The Gulp api
  1. gulp.task(name [ , deps], fn)
  2. gulp.src(globs)
  3. gulp.dest(path)
  4. gulp.watch(glob [, opts], tasks)
gulp.task(name[, deps], fn) 
gulp.task('somename', function() {
	// Do stuff
});
gulp.task('somename',['test', 'task2'], function() {
	// Do stuff
});
gulp.task('somename',['test', 'task2']);
> gulp somename

let's try it in Real World

کامپایل کردن فایل های Sass

var gulp = require('gulp');
var sass = require('gulp-sass');

gulp.task('css', function(){
	return gulp.src('src/assets/*.scss')
		.pipe(sass())
		.pipe(gulp.dest('build/assets/css'))
});

sourcemaps

var gulp = require('gulp');
var sass = require('gulp-sass');
var sourcemaps = require('gulp-sourcemaps');

gulp.task('css', function(){
	return gulp.src('src/assets/*.scss')
		.pipe(sourcemaps.init())
		.pipe(sass())
		.pipe(sourcemaps.write('.'))
		.pipe(gulp.dest('build/assets/css'))
});

autoprefixer

var gulp = require('gulp');
var sass = require('gulp-sass');
var sourcemaps = require('gulp-sourcemaps');
var autoprefixer = require('gulp-autoprefixer');

gulp.task('css', function(){
	return gulp.src('src/assets/*.scss')
		.pipe(sourcemaps.init())
		.pipe(sass())
		.pipe(autoprefixer({
			browsers: ['last 5 versions', '> 5%'],
			cascade: false
		}))
		.pipe(sourcemaps.write('.'))
		.pipe(gulp.dest('build/assets/css'))
});

minifyCSS

var gulp = require('gulp');
var sass = require('gulp-sass');
var sourcemaps = require('gulp-sourcemaps');
var autoprefixer = require('gulp-autoprefixer');
var cssmin = require('gulp-cssmin');

gulp.task('css', function(){
	return gulp.src('src/assets/*.scss')
		.pipe(sourcemaps.init())
		.pipe(sass())
		.pipe(autoprefixer({
			browsers: ['last 5 versions', '> 5%'],
			cascade: false
		}))
		.pipe(cssmin())
		.pipe(sourcemaps.write('.'))
		.pipe(gulp.dest('build/assets/css'))
});

js concat & Babel

var gulp = require('gulp');
var sourcemaps = require('gulp-sourcemaps');
var concat = require('gulp-concat');
var babel = require('gulp-babel');

gulp.task('js', function(){
    return gulp.src('src/assets/js/*.js')
        .pipe(sourcemaps.init())
		.pipe(concat('app.js'))
		.pipe(babel({
			presets: ['env']
		}))
        .pipe(sourcemaps.write('.'))
        .pipe(gulp.dest('build/assets/js'))
});

watch

gulp.task('watch', ['default'], function() {
	gulp.watch([
		'./src/**/*.scss'
	], ['css']);

	gulp.watch([
		'./src/**/*.js'
	], ['js']);
});

BrowserSyunc + GulpJs

var browserSync = require('browser-sync').create();

gulp.task('css', function(){
	return gulp.src('src/assets/sass/*.scss')
		.pipe(sass())
		.pipe(gulp.dest('build/assets/css'))
		.pipe(browserSync.stream())
});

gulp.task('serve', ['default'], function() {
	browserSync.init({
		server: "./build" // static server
		// proxy: "localhost/wordpress" // dynamic server
	});
	gulp.watch("src/assets/sass/*.scss", ['css']);
	gulp.watch("src/**/*.html", ['copy'])
	.on('change', browserSync.reload);
});
